99  def feed(self, line):
100  """ Given a line, determine the interaction type and the atoms involved (the suffix).
101 
102  TINKER generally has stuff like this:
103 
104  @verbatim
105  bond-cubic -2.55
106  bond-quartic 3.793125
107 
108  vdw 1 3.4050 0.1100
109  vdw 2 2.6550 0.0135 0.910 # PRM 4
110 
111  multipole 2 1 2 0.25983
112  -0.03859 0.00000 -0.05818
113  -0.03673
114  0.00000 -0.10739
115  -0.00203 0.00000 0.14412
116  @endverbatim
117 
118  The '#PRM 4' has no effect on TINKER but it indicates that we
119  are tuning the fourth field on the line (the 0.910 value).
120 
121  @todo Put the rescaling factors for TINKER parameters in here.
122  Currently we're using the initial value to determine the
123  rescaling factor which is not very good.
124 
125  Every parameter line is prefaced by the interaction type
126  except for 'multipole' which is on multiple lines. Because
127  the lines that come after 'multipole' are predictable, we just
128  determine the current line using the previous line.
129 
130  Random note: Unit of force is kcal / mole / angstrom squared.
131 
132  """
133  s = line.split()
134  self.ln += 1
135  # No sense in doing anything for an empty line or a comment line.
136  if len(s) == 0 or match('^#',line): return None, None
137  # From the line, figure out the interaction type. If this line doesn't correspond
138  # to an interaction, then do nothing.
139  if s[0].upper() in pdict:
140  self.itype = s[0].upper()
141  # This is kind of a silly hack that allows us to take care of the 'multipole' keyword,
142  # because of the syntax that the TINKER .prm file uses.
143  elif s[0].upper() == 'MULTIPOLE':
144  self.itype = 'MCHARGE'
145  elif self.itype == 'MCHARGE':
146  self.itype = 'DIPOLE'
147  elif self.itype == 'DIPOLE':
148  self.itype = 'QUADX'
149  elif self.itype == 'QUADX':
150  self.itype = 'QUADY'
151  elif self.itype == 'QUADY':
152  self.itype = 'QUADZ'
153  else:
154  self.itype = None
155 
156  if self.itype in pdict:
157  if 'Atom' in pdict[self.itype]:
158  # List the atoms in the interaction.
159  self.atom = [s[i] for i in pdict[self.itype]['Atom']]
160  # The suffix of the parameter ID is built from the atom #
161  # types/classes involved in the interaction.

533 
534  """ Optimize the geometry and align the optimized geometry to the starting geometry. """
535

820  def molecular_dynamics(self, nsteps, timestep, temperature=None, pressure=None, nequil=0, nsave=1000, minimize=True, anisotropic=False, threads=1, verbose=False, **kwargs):
821 
822  """
823  Method for running a molecular dynamics simulation.
824 
825  Required arguments:
826  nsteps = (int) Number of total time steps
827  timestep = (float) Time step in FEMTOSECONDS
828  temperature = (float) Temperature control (Kelvin)
829  pressure = (float) Pressure control (atmospheres)
830  nequil = (int) Number of additional time steps at the beginning for equilibration
831  nsave = (int) Step interval for saving and printing data
832  minimize = (bool) Perform an energy minimization prior to dynamics
833  threads = (int) Specify how many OpenMP threads to use
834 
835  Returns simulation data:
836  Rhos = (array) Density in kilogram m^-3
837  Potentials = (array) Potential energies
838  Kinetics = (array) Kinetic energies
839  Volumes = (array) Box volumes
840  Dips = (3xN array) Dipole moments
841  EComps = (dict) Energy components
842  """
843 
844  md_defs = OrderedDict()
845  md_opts = OrderedDict()
846  # Print out averages only at the end.
847  md_opts["printout"] = nsave
848  md_opts["openmp-threads"] = threads
849  # Langevin dynamics for temperature control.
850  if temperature != None:
851  md_defs["integrator"] = "stochastic"
852  else:
853  md_defs["integrator"] = "beeman"
854  md_opts["thermostat"] = None
855  # Periodic boundary conditions.
856  if self.pbc:
857  md_opts["vdw-correction"] = ''
858  if temperature != None and pressure != None:
859  md_defs["integrator"] = "beeman"
860  md_defs["thermostat"] = "bussi"
861  md_defs["barostat"] = "montecarlo"
862  if anisotropic:
863  md_opts["aniso-pressure"] = ''
864  elif pressure != None:
865  warn_once("Pressure is ignored because temperature is turned off.")
866  else:
867  if pressure != None:
868  warn_once("Pressure is ignored because pbc is set to False.")
869  # Use stochastic dynamics for the gas phase molecule.
870  # If we use the regular integrators it may miss
871  # six degrees of freedom in calculating the kinetic energy.
872  md_opts["barostat"] = None
873 
874  eq_opts = deepcopy(md_opts)
875  if self.pbc and temperature != None and pressure != None:
876  eq_opts["integrator"] = "beeman"
877  eq_opts["thermostat"] = "bussi"
878  eq_opts["barostat"] = "berendsen"
879 
880  if minimize:
881  if verbose: logger.info("Minimizing the energy...")
882  self.optimize(method="bfgs", crit=1)
883  os.system("mv %s.xyz_2 %s.xyz" % (self.name, self.name))
884  if verbose: logger.info("Done\n")
885 
886  # Run equilibration.
887  if nequil > 0:
888  write_key("%s-eq.key" % self.name, eq_opts, "%s.key" % self.name, md_defs)
889  if verbose: printcool("Running equilibration dynamics", color=0)
890  if self.pbc and pressure != None:
891  self.calltinker("dynamic %s -k %s-eq %i %f %f 4 %f %f" % (self.name, self.name, nequil, timestep, float(nsave*timestep)/1000,
892  temperature, pressure), print_to_screen=verbose)
893  else:
894  self.calltinker("dynamic %s -k %s-eq %i %f %f 2 %f" % (self.name, self.name, nequil, timestep, float(nsave*timestep)/1000,
895  temperature), print_to_screen=verbose)
896  os.system("rm -f %s.arc" % (self.name))
897 
898  # Run production.
899  if verbose: printcool("Running production dynamics", color=0)
900  write_key("%s-md.key" % self.name, md_opts, "%s.key" % self.name, md_defs)
901  if self.pbc and pressure != None:
902  odyn = self.calltinker("dynamic %s -k %s-md %i %f %f 4 %f %f" % (self.name, self.name, nsteps, timestep, float(nsave*timestep/1000),
903  temperature, pressure), print_to_screen=verbose)
904  else:
905  odyn = self.calltinker("dynamic %s -k %s-md %i %f %f 2 %f" % (self.name, self.name, nsteps, timestep, float(nsave*timestep/1000),
906  temperature), print_to_screen=verbose)
907 
908  # Gather information.
909  os.system("mv %s.arc %s-md.arc" % (self.name, self.name))
910  self.mdtraj = "%s-md.arc" % self.name
911  edyn = []
912  kdyn = []
913  temps = []
914  for line in odyn:
915  s = line.split()
916  if 'Current Potential' in line:
917  edyn.append(float(s[2]))
918  if 'Current Kinetic' in line:
919  kdyn.append(float(s[2]))
920  if len(s) > 0 and s[0] == 'Temperature' and s[2] == 'Kelvin':
921  temps.append(float(s[1]))
922 
923  # Potential and kinetic energies converted to kJ/mol.
924  edyn = np.array(edyn) * 4.184
925  kdyn = np.array(kdyn) * 4.184
926  temps = np.array(temps)
927 
928  if verbose: logger.info("Post-processing to get the dipole moments\n")
929  oanl = self.calltinker("analyze %s-md.arc" % self.name, stdin="G,E,M", print_to_screen=False)
930 
931  # Read potential energy and dipole from file.
932  eanl = []
933  dip = []
934  mass = 0.0
935  ecomp = OrderedDict()
936  havekeys = set()
937  first_shot = True
938  for ln, line in enumerate(oanl):
939  strip = line.strip()
940  s = line.split()
941  if 'Total System Mass' in line:
942  mass = float(s[-1])
943  if 'Total Potential Energy : ' in line:
944  eanl.append(float(s[4]))
945  if 'Dipole X,Y,Z-Components :' in line:
946  dip.append([float(s[i]) for i in range(-3,0)])
947  if first_shot:
948  for key in eckeys:
949  if strip.startswith(key):
950  if key in ecomp:
951  ecomp[key].append(float(s[-2])*4.184)
952  else:
953  ecomp[key] = [float(s[-2])*4.184]
954  if key in havekeys:
955  first_shot = False
956  havekeys.add(key)
957  else:
958  for key in havekeys:
959  if strip.startswith(key):
960  if key in ecomp:
961  ecomp[key].append(float(s[-2])*4.184)
962  else:
963  ecomp[key] = [float(s[-2])*4.184]
964  for key in ecomp:
965  ecomp[key] = np.array(ecomp[key])
966  ecomp["Potential Energy"] = edyn
967  ecomp["Kinetic Energy"] = kdyn
968  ecomp["Temperature"] = temps
969  ecomp["Total Energy"] = edyn+kdyn
970 
971  # Energies in kilojoules per mole
972  eanl = np.array(eanl) * 4.184
973  # Dipole moments in debye
974  dip = np.array(dip)
975  # Volume of simulation boxes in cubic nanometers
976  # Conversion factor derived from the following:
977  # In [22]: 1.0 * gram / mole / (1.0 * nanometer)**3 / AVOGADRO_CONSTANT_NA / (kilogram/meter**3)
978  # Out[22]: 1.6605387831627252
979  conv = 1.6605387831627252
980  if self.pbc:
981  vol = np.array([BuildLatticeFromLengthsAngles(*[float(j) for j in line.split()]).V \
982  for line in open("%s-md.arc" % self.name).readlines() \
983  if (len(line.split()) == 6 and isfloat(line.split()[1]) \
984  and all([isfloat(i) for i in line.split()[:6]]))]) / 1000
985  rho = conv * mass / vol
986  else:
987  vol = None
988  rho = None
989  prop_return = OrderedDict()
990  prop_return.update({'Rhos': rho, 'Potentials': edyn, 'Kinetics': kdyn, 'Volumes': vol, 'Dips': dip, 'Ecomps': ecomp})
991  return prop_return
