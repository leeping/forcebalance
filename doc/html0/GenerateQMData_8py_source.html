00032 def even_list(totlen, splitsize):
00033     joblens = np.zeros(splitsize,dtype=int)
00034     subsets = []
00035     for i in range(totlen):
00036         joblens[i%splitsize] += 1
00037     jobnow = 0
00038     for i in range(splitsize):
00039         subsets.append(range(jobnow, jobnow + joblens[i]))
00040         jobnow += joblens[i]
00041     return subsets
